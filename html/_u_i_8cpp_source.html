624vector<string> UI::airport(string code, const FileReader& fileReader) {
625 if (code.size() == 3 && std::all_of(code.begin(), code.end(), ::isupper)) {
626 vector<string> airportCities;
627 if (fileReader.get_Airports().find(code)!=fileReader.get_Airports().end()) {
628 airportCities.push_back(code);
629 return airportCities;
630 }
631 else{
632 return {};
633 }
634 }
635 else if(existes_airport_from_name(code, fileReader))
636 {
637 vector<string> airportCities;
638 airportCities.push_back(get_airport_code(code, fileReader));
639 return airportCities;
640 }
641 else if (fileReader.get_Airports_Cities().find(code) != fileReader.get_Airports_Cities().end()) {
642 vector<string> airportCities = Operations::get_Airport_CityName(code, fileReader);
643 if(airportCities.empty()){
644 return {};
645 }
646 return airportCities;
647 }
648 // Check if code follows the pattern for Coordinates (two numbers)
649 else if (std::count(code.begin(), code.end(), ',') == 1) {
650 float latitude, longitude;
651 if (sscanf(code.c_str(), "%f,%f", &latitude, &longitude) == 2) {
652 string airportCoordinates = Operations::get_Airport_Coordinates(latitude, longitude, fileReader);
653 if(airportCoordinates.empty()){
654 return {};
655 }
656 vector<string> airportCities;
657 airportCities.push_back(airportCoordinates);
658 return airportCities;
659 }
660 }
661
662 return {};
663}
